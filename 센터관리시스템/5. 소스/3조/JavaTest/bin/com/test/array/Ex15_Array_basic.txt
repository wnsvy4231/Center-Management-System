	private static void m10() {

		//성적 데이터
		// - 학생 3명
		// - 과목 3과목
		// - 점수 0~100
		// - 총점, 평균
		
		int count=3;
		
		String[] name = new String[count];
		int[][] score = new int[count][3];
		
		name[0]="홍길동";
		name[1]="아무개";
		name[2]="하하하";
		
		for (int i = 0; i < score.length; i++) {			//학생
			for (int j = 0; j < score[0].length; j++) {		//과목
				score[i][j]=(int)(Math.random()*41)+60;	//60~100
			}
		}
		
		//성적표 출력
		System.out.println("==============================================");
		System.out.println("                  성적표");
		System.out.println("==============================================");
		System.out.println("[이름]\t[국어]\t[영어]\t[수학]\t[총점]\t[평균]");
		
		for (int i = 0; i < score.length; i++) {
			System.out.printf("%s\t", name[i]);
			
			int sum=0;
			
			for (int j = 0; j < score.length; j++) {
				System.out.printf("%4d\t",score[i][j]);
				sum+=score[i][j];
			}
//			System.out.printf("%d\t", score[i][0]);
//			System.out.printf("%d\t", score[i][1]);
//			System.out.printf("%d\t", score[i][2]);
			System.out.printf("%4d\t %4.1f",sum, sum/3.0);
			System.out.println();
		}
		
	}

	private static void m7() {
		
		//다차원 배열
		// - 배열은 차수를 가진다.
		// - 1차원 배열
		// - 2차원 배열
		// - 3차원 배열
		
		//1차원 배열 탐색 -> 단일 for문
		int[] num1 = new int[3];
		
		//2차원 배열 탐색 -> 이중 for문
		int[][] num2 = new int[2][3];	//2행3열
		
		num2[0][0]=100;
		num2[0][1]=200;
		num2[0][2]=300;
		
		num2[1][0]=400;
		num2[1][1]=500;
		num2[1][2]=600;
		
		//System.out.println(num2.length);
		
		for (int i = 0; i < num2.length; i++) {		//행, 층
			for (int j = 0; j < num2[0].length; j++) {	//열,호수
				//System.out.println(i+", "+j);
				System.out.printf("num2[%d][%d] = %d\n",i,j,num2[i][j]);
			}
		}
		
		//자료형?
		//1. num2 -> int[][] -> int 2차원 배열
		//2. num2[0][0] -> int
		//3. num2[0] -> int[] -> int 1차원 배열(***)
		
		//3차원
		int[][][] num3 = new int[2][2][3];
		
		for (int i = 0; i < 2; i++) {
			for (int j = 0; j < 2; j++) {
				for (int k = 0; k < 3; k++) {
					num3[i][j][k]=i*j*k;
				}
			}
		}
		
		for (int i = 0; i < 2; i++) {
			for (int j = 0; j < 2; j++) {
				for (int k = 0; k < 3; k++) {
					System.out.printf("%d\n",num3[i][j][k]);
				}
			}
		}
		
		//1차원 배열에만 사용
		System.out.println(Arrays.toString(num1));	
		
	}//m7

	private static void m3() {

		//배열 탐색
		String[] names = new String[5];
		
		names[0]="홍길동";
		names[1]="아무개";
		names[2]="유재석";
		names[3]="강호동";
		names[4]="신동엽";
		
		//순차 탐색	- for문
		for (int i = 0; i < names.length; i++) {
			System.out.println(names[i]);
		}
		System.out.println();
		
		//순차 탐색 - 향상된 for문
		// - for (변수 : 배열) {}
		// - 무조건 ~ 끝방까지 탐색
		for (String name : names) {
			System.out.println(name);
		}
		System.out.println();
		
		//배열 데이터 가공
		for (int i = 0; i < names.length; i++) {
			names[i]+="님";
			System.out.println(names[i]);
		}
		System.out.println();
		
		// - 읽기 전용(배열의 요소를 수정할 수 없다.)
		for (String name : names) {
			//name+="씨";
			System.out.println(name);
		}
		System.out.println();
		
	}
